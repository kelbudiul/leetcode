int countKDifference(int* nums, int numsSize, int k){
    int counter=0;
    unsigned int abs = 0;
    int u = 0;
    for(int i = 0; i < numsSize; i++){
        for(int j = 0; j < numsSize; j++){
            u = nums[i] - nums[j];
            abs = ((unsigned)((u<0)?-u:u));
            if(i < j && abs == k)
                counter++;
                abs = 0;
                u = 0;
        }
    }

    return counter;
}
