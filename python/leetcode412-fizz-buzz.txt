class Solution:
    def fizzBuzz(self, n: int) -> List[str]:
        j = 1
        answer = []
        for i in range(n):
            if j % 3 == 0 and j % 5 == 0:
                answer.append("FizzBuzz")
            elif j % 3 == 0:
                answer.append("Fizz")
            elif j % 5 == 0:
                answer.append("Buzz")
            else:
                answer.append(str(j))
            j += 1

        return answer
